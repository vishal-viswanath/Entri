#include <stdio.h>

int main()
{
    int arr[5] = {10,20,30,40,50};
    
    int *ptr = arr;
    
    for(int i=0;i<5;i++)
    {
        printf("value of arr[%d] = %d and address is %p\n",i,*(ptr+i),(ptr+i));
        
    }
    
    printf("address of ptr: %p and address of arr[0]: %p",&ptr,&ptr[0]);
}
