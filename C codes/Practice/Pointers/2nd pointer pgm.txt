#include <stdio.h>

int main()
{
    int a = 10;
    int b = 20;
    int c = 30;
    
    int *ptr1 = &a;
    int *ptr2 = &b;
    int *ptr3 = &c;
    
    printf("value of a: %d\n",a);
    printf("value of a, using pointer: %d\n",*ptr1);
    printf("address of a: %p\n",&a);
    printf("address of a, using pointer: %p\n",ptr1);
    printf("address of ptr1 is: %p\n\n",&ptr1);
    
    printf("value of b: %d\n", b);
    printf("value of b, using pointer: %d\n",*ptr2);
    printf("address of b: %p\n",&b);
    printf("address of b, using pointer: %p\n",ptr2);
    printf("address of ptr2 is: %p\n\n",&ptr2);
    
    printf("value of c: %d\n",c);
    printf("value of c, using pointer: %d\n",*ptr3);
    printf("address of c: %p\n",&c);
    printf("address of c, using pointer: %p\n", ptr3);
    printf("address of ptr3: %p\n\n",&ptr3);
    
    printf("%zu",sizeof(ptr1));
    printf("%zu",sizeof(a));
    
}
