#include <stdio.h>

int main()
{
    int a = 10;
    int *ptr;
    ptr = &a;
    
    printf("value of a is : %d\n",a);
    printf("address of a is: %p\n",&a);
    printf("value of the address stored inside the pointer: %d\n",*ptr);
    printf("enter the address of a using pointer: %p\n",ptr);
    
    printf("address of pointer variable is: %p\n",&ptr);
    
    
}
