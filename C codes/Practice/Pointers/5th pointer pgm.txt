#include <stdio.h>


             /*Add 5 with each array value*/

int main()
{
    int arr[5] = {1,2,3,4,5};
    int *ptr;
    ptr = arr;
    
    for(int i = 0; i<5; i++)
    {
        printf("value of arr[%d] = %d and address is %p\n",i,*(ptr+i),ptr+i);
        printf("value of arr[%d] = %d and address is %p\n\n",i,*(ptr+i)+5,ptr+i);
    }
    
}
