#include <stdio.h>
#include <stdlib.h>

int main()
{
    int *ptr;
    int n;
    
    printf("enter the number of integers to be stored: ");
    scanf("%d",&n);
    
    ptr = (int*)malloc(n * sizeof(int));
    
    if(ptr == NULL)
    {
        printf("memory allocation failed");
        return 1;
    }
    
    for(int i=0; i<n; i++)
    {
        printf("enter the %d integer: ",i+1);
        scanf("%d",ptr+i);
    }
    
    for(int i = 0; i<n; i++)
    {
        printf("the value of %d integer is: %d\n",i+1,*ptr+i);
    }
    free(ptr);
    return 0;
}
