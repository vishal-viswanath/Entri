#include <stdio.h>

void printbin(int n)
{
    int mask;
    
    for(int i = 31; i >= 0; i--)
    {
        mask = 1 << i;
        
        if(n & mask)
        {
        printf("1");
        }
        else
        {
            printf("0");
        }
    }
}

int main()
{
int a;

printf("Enter a value: ");
scanf("%d",&a);

printbin(a);

}
