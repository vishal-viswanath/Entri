#include <stdio.h>

int main()
{
   int arr[5] = {1,2,3,4,5};
   int* ptr = arr;

       printf("value of arr[%d]=%d and the memmory address is %p\n",i,*(ptr+i),ptr+i);

       printf("value of arr[%d]=%d and the memmory address is %p\n\n",i,(*(ptr+i)+5),ptr+i);
