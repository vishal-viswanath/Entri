#include <stdio.h>

char print_bin(char num)
{
    int mask;
    for(int i = 7; i >= 0; i--)
    {
        mask = 1<<i;
    
        if(num & mask)
        {
          
           printf("1");
            
        }
        else
        {
           printf("0");
        }
    
    } 
    
    printf("\n");
    
}

char set_bit(char value)
{
    int mask;
    
    mask = 1<<3;
    
    value = value|mask;
    
    print_bin(value);
}

char clear_bit(char value)
{
    int mask;
    mask = ~(1<<6);
    
    value = value & mask;
    
    print_bin(value);
    
}

char toggle_bit(char value)
{
    int mask,value1;
    mask = 1 << 0;
    value1 = value ^ mask;
    
    
    print_bin(value1);
    
    return value1;
    
}

int main()
{
    
    char value;
    int back;
    
    printf("enter a unsigned char: ");
    scanf("%c",&value);
    
    print_bin(value);
    set_bit(value);
    clear_bit(value);
    back = toggle_bit(value);
    
    printf("new value of %d",back);
    
    
}
