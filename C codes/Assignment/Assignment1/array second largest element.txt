#include <stdio.h>

int sec_lar_value(int n1,int *ptr1)
{
    int largest,sec_largest;
    
    largest = ptr1[0];
    sec_largest= ptr1[0];

   
    for(int i=0; i<n1; i++)
    {
        if(largest < *(ptr1+i))
        {
            sec_largest = largest;
            largest = *(ptr1+i);
        }
        else if(*(ptr1+i) < largest && *(ptr1+i) > sec_largest)
        {
            sec_largest=*(ptr1+i);
        }
    }
    
    if(sec_largest == ptr1[0])
        {
            sec_largest = 0;
        }
    
    return sec_largest;

}
